bool Convert(unsigned int number, unsigned int toBase, char * result, unsigned int resultMaxSize)
{
    if(toBase<2 || toBase>16)return false;
    int cont=-1;
    while(number!=0)
    {
        int rest=number%toBase;
        if(cont+1>resultMaxSize)return false;
            else
            {
                cont++;
                if(rest>=10)result[cont]=65+rest-10;
                    else
                        {
                            char a[100];
                            itoa(rest,a,10);
                            result[cont]=a[0];
                        }
            }
        number=number/toBase;
    }
    cont++;
    result[cont]=NULL;
    strrev(result);
    return true;
}
